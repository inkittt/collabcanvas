Issue 1: Canvas State Structure

The canvas.toJSON() might not include element IDs properly.

// Check if canvas objects have IDs: console.log('Canvas state:', canvas.toJSON(['id'])); // Should show objects with id property
Issue 2: Database Sync Function Not Called

The syncCanvasStateWithDatabase function might not be executing.

Issue 3: Element ID Mismatch

Canvas objects might have different IDs than database elements.

// Check element IDs: console.log('Canvas element IDs:', canvas.getObjects().map(obj => obj.id)); console.log('Database element IDs:', elements.map(el => el.id));
Issue 4: Real-time Interference

Real-time updates might be interfering with undo/redo sync.

// Temporarily disable real-time during undo/redo: setIsUserModifying(true); // Disable real-time updates await syncCanvasStateWithDatabase(prevState, canvasId); setTimeout(() => setIsUserModifying(false), 1000); // Re-enable

Here's a more robust implementation that should work:

// Better undo function with detailed logging const handleUndo = useCallback(async () => { if (!fabricCanvasRef.current || undoStack.length === 0) return; console.log('üîÑ UNDO: Starting undo operation'); const canvas = fabricCanvasRef.current; const currentState = canvas.toJSON(['id']); console.log('üìä Current canvas state:', currentState); console.log('üìö Undo stack length:', undoStack.length); setRedoStack(prev => [...prev, currentState]); const prevState = undoStack[undoStack.length - 1]; setUndoStack(prev => prev.slice(0, -1)); console.log('‚è™ Restoring to previous state:', prevState); // Disable real-time updates during sync setIsUserModifying(true); canvas.loadFromJSON(prevState, async () => { canvas.renderAll(); console.log('üé® Canvas restored visually'); try { console.log('üíæ Starting database sync...'); // Get current database elements const dbElements = await ElementService.getCanvasElements(canvasId); console.log('üìã Database elements:', dbElements.map(el => el.id)); // Get canvas elements const canvasObjects = canvas.getObjects(); console.log('üéØ Canvas objects:', canvasObjects.map(obj => obj.id)); // Delete elements not in canvas const canvasIds = new Set(canvasObjects.map(obj => obj.id).filter(id => id)); const elementsToDelete = dbElements.filter(el => !canvasIds.has(el.id)); console.log('üóëÔ∏è Elements to delete:', elementsToDelete.map(el => el.id)); for (const element of elementsToDelete) { console.log('üóëÔ∏è Deleting element:', element.id); await ElementService.deleteElement(element.id); } console.log('‚úÖ Database sync completed'); } catch (error) { console.error('‚ùå Database sync failed:', error); } finally { // Re-enable real-time updates setTimeout(() => setIsUserModifying(false), 1000); } }); }, [undoStack, canvasId]);
